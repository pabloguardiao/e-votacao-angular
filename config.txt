	function convertDatePtBr(dataPtBr) {
		if(dataPtBr) {
			try {
				const data = moment(dataPtBr+' 12', 'DD/MM/YYYY HH');
				
				if (!data.isValid()) {
					throw Error();
				}

				return new Date(data.format());
			} catch (err) {
				console.error(err);
				return null;
			}
		} else {
			return null;
		}
